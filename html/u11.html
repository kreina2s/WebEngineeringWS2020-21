./src/globals.js
export var menu,
  jsonContent = 0;

load().then((val) => {
  jsonContent = val;
});

async function load() {
  var file = await fetch("topics.json");
  var result = await file.json();
  console.log(result);
  return result;
}

export function initButtons(x) {
  var navigation = document.getElementById("nav");
  navigation.innerHTML = "";
  menu = jsonContent[x];
  for (let key in menu) {
    var button = document.createElement("button");
    button.innerHTML = key;
    button.onclick = () => initTopic(key);
    navigation.appendChild(button);
  }
}

export function initTopic(x) {
  // Clean up main first
  var main = document.getElementById("main");
  main.innerHTML = "";

  // Append new text paragraphs
  var p = document.createElement("p");
  var ref = document.createElement("p");
  var a = document.createElement("a");
  a.innerHTML = menu[x].references;
  a.href = menu[x].references;

  ref.appendChild(document.createTextNode("Reference: "));
  p.appendChild(document.createTextNode(menu[x].content));
  main.appendChild(p);
  main.appendChild(ref);
  main.appendChild(a);
}
